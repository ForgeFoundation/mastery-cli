Get Started

Installation

npm install ervy

Usage

const ervy = require('ervy')
const { bar, pie, bullet, donut, gauge, scatter } = ervy
        
// Prepare data to render chart
const data = [
  { key: 'A', value: 30 },
  { key: 'B', value: 10 },
  { key: 'C', value: 50 }
]

// Then use any chartType you like, for example:
console.log(
  bar(data, [options])
)
